\begin{Description}
Core modeling, preprocessing, and utility functions provided by VectorForgeML.
\end{Description}
%
\begin{Details}
The package provides a reference-class linear regression model, feature preprocessing helpers,
and pipeline composition utilities with an Rcpp backend.
\end{Details}
%
\begin{Section}{Model API}
\code{LinearRegression} supports \code{\$fit(X, y)} and \code{\$predict(X)}.
\code{fit\_linear\_model()} and \code{predict\_linear\_model()} are convenience wrappers.
\end{Section}
%
\begin{Section}{Preprocessing API}
\code{StandardScaler}, \code{MinMaxScaler}, \code{LabelEncoder}, \code{OneHotEncoder},
and \code{drop\_constant\_columns()} (C++-backed).
\end{Section}
%
\begin{Section}{Utilities}
\code{Pipeline}, \code{ColumnTransformer}, \code{train\_test\_split},
\code{mse}, \code{rmse}, and \code{r2\_score}.
\end{Section}
